      async function initMap() {

        // Center on Carlisle, MA

        const map = new google.maps.Map(document.getElementById("map"), {

          center: { lat: 42.529, lng: -71.35 },

          zoom: 14,

          mapTypeId: "roadmap",

        });


        // URL of your public KML file in Google Drive

        const kmlUrl = "https://drive.google.com/uc?export=download&id=17uU8PbDkT_S0sYOHvdY3EHxWP9l-YLGK";


        // Load the KML layer

        const kmlLayer = new google.maps.KmlLayer({

          url: kmlUrl,

          map: map,

          preserveViewport: false,

          suppressInfoWindows: false,

        });


        kmlLayer.addListener("status_changed", () => {

          if (kmlLayer.getStatus() !== "OK") {

            alert("Could not load map data. Make sure the KML file is shared publicly.");

          }

        });

      }

      window.onload = initMap;
